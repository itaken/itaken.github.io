scss 一些语法

css scss
开发日常
Publish Date:   2017-08-10

  1. @import 引入其他样式

    @import "home";
// 引入 home.scss 样式

@import "home.css"
// 引入 home.css 样式

  2. 声明变量

    // 定义支持的浏览器
$supported-browsers: chrome,firefox,ie,opera,safari;

// 红色
$red-color: #F00;

.lantern &#123;
    color: $red-color;
    border: 1px solid $red-color;
&#125;
// .lantern&#123; color: #F00; border: 1px solid #F00; &#125;

  3. 样式嵌套

    .navbar&#123;
    position: fixed;
    width: 100%;
    .navbar-header&#123;
        float: left;
        a&#123;
            font-size: 24px;
        &#125;
    &#125;
&#125;
// .navbar .navbar-header a &#123; font-size: 24px; &#125;

.container &#123;
    h1, h2, h3 &#123;
        margin-bottom: 6px;
    &#125;
&#125;
// .container h1, .container h2, .container h3 &#123; margin-bottom: 6px; &#125;

  4. &父选择器

    ul&#123;
    list-style: none;
    li&#123;
        background-color: #37bed4;
        &:hover&#123;
            background-color: #fffdf3;
        &#125;
    &#125;
&#125;
// ul li:hover &#123; background-color: #fffdf3; &#125;

  5. 子组合选择器和同层组合选择器：>、+和~

    • 子组合选择器>, 选择一个元素的直接子元素, 例如: article > section { border: 1px solid #ccc }
    • 同层相邻组合选择器+ , 例如: header + p { font-size: 1.1em }
    • 同层全体组合选择器~ , 例如: article ~ article { border-top: 1px dashed #ccc }

  6. @mixin混合器

    @mixin rounded-corners &#123;
    -moz-border-radius: 5px;
    -webkit-border-radius: 5px;
    border-radius: 5px;
&#125;

notice &#123;
    @include rounded-corners;
&#125;

@mixin link-colors($normal)&#123;
    color: $normal;
&#125;
// 有传参情况, 使用 `@include link-colors(red)` 引用

  7. 选择器继承, @extend

    .admin-content&#123;
    height: 100%;
&#125;
.admin-sidebar &#123;
    @extend .admin-content;
&#125;
// .admin-sidebar 同 .admin-content 样式一样

%non-margin-pading &#123;
    margin: 0px;
    padding: 0px;
&#125;
.am-u-md-5,.am-u-md-7,.am-u-md-6&#123;
    @extend %non-margin-pading;
&#125;
// 同上, 但是 non-margin-pading 并非作为一个样式

  8. @at-root 根样式

    .am-u-md-5&#123;
    @at-root&#123;
        .am-tab-panel&#123;
            margin: 0 4px 0 4px;
        &#125;
    &#125;
&#125;
// .am-tab-panel 将作为第一级样式, (与 .am-u-md-5 同级)
